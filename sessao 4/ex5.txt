using System.Security.Cryptography;

namespace atividade_sessao_logica_part2_c_
{
    class Program
    {
        static void Main(string[] args)
        {

            Apresentar();

            int mes = PedirDados();

            string resul = Verificacao(mes);

            Console.WriteLine(resul);
        }


        public static void Apresentar()
        {
            Console.WriteLine("PROGRAMA DOS MESES");
        }

        public static int PedirDados()
        {
            Console.WriteLine("Me informe o mês em número (exemplo: 2 = fevereiro)");

            int mes = Convert.ToInt32(Console.ReadLine());

            return mes;
        }

        public static string Verificacao(double mes)
        {

            string resp;

            if (mes == 1)
            {
                resp = "O mês que você selecionou é janeiro";
            }
            else if (mes == 2)
            {
                resp = "O mês que você selecionou é fevereiro";
            }
            else if (mes == 3)
            {
                resp = "O mês que você selecionou é março";
            }
            else if (mes == 4)
            {
                resp = "O mês que você selecionou é abril";
            }
            else if (mes == 5)
            {
                resp = "O mês que você selecionou é maio";
            }
            else if (mes == 6)
            {
                resp = "O mês que você selecionou é junho";
            }
            else if (mes == 7)
            {
                resp = "O mês que você selecionou é julho";
            }
            else if (mes == 8)
            {
                resp = "O mês que você selecionou é agosto";
            }
            else if (mes == 9)
            {
                resp = "O mês que você selecionou é setembro";
            }
            else if (mes == 10)
            {
                resp = "O mês que você selecionou é outubro";
            }
            else if (mes == 11)
            {
                resp = "O mês que você selecionou é novembro";
            }
            else if (mes == 12)
            {
                resp = "O mês que você selecionou é dezembro";
            }
            else
            {
                resp = "Mês invalido";
            }

            return resp;
        }
    }
}
